1. write a program that reads 5 random 3 digit vlaues
   and then outputs the minimum and the maximum value.
Ans)
CODE:
#!/bin/bash -x

read -a number

max=${number[0]}
min=${number[0]}

for i in "${number[@]}"
do
    # It will Update max if applicable
    if [[ "$i" -gt "$max" ]] 
    then
        max="$i"
    fi

    # It will Update min if applicable
    if [[ "$i" -lt "$min" ]] 
    then
        min="$i"
    fi
done

echo "Maximum Value is: $max"
echo "Minimum Value is: $min"

echo "3. Feet To Meter."
echo "4. Meter To Feet."

read input


InchesToFeet=1
FeetToInches=2
FeetToMeter=3
MeterToFeet=4

case $input in
         $InchesToFeet)
            echo "Enter Inches"
            read inches
            outputone=$(($inches / 12))
            echo "Inches To Feet(ft) is:" $outputone
               ;;

               ;;
         *)
            output=InvalidRangeNumber
               ;;
esac
